131 public void ApplyRecoil(Node weaponNode, bool isFlipped)
132 {
133 // Cancela el tween anterior si aún está activo.
134 if (_recoilTween != null && _recoilTween.IsValid())
135 _recoilTween.Kill();
136
137 float recoilRad = Mathf.DegToRad(isFlipped ? RecoilAngleDeg : -RecoilAngleDeg);

139 _recoilTween = weaponNode.CreateTween();
140 _recoilTween
141 .TweenProperty(weaponNode, "rotation", recoilRad, RecoilKickDuration)
142 .SetEase(Tween.EaseType.Out)
143 .SetTrans(Tween.TransitionType.Quad);
144 _recoilTween
145 .TweenProperty(weaponNode, "rotation", 0f, RecoilReturnDuration)
146 .SetEase(Tween.EaseType.In)
147 .SetTrans(Tween.TransitionType.Quad);
