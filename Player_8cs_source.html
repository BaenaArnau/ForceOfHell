177 private void UpdateSpriteDirection()
178 {
179 if (_animatedSprite == null)
180 return;
181
182 var currentFlip = _animatedSprite.FlipH;
183 var nextFlip = Velocity.X switch
184 {
185 < 0f => true,
186 > 0f => false,
187 _ => currentFlip
188 };
189
190 // Evita asignaciones repetidas si no cambia la dirección.
191 if (currentFlip == nextFlip)
192 return;
193
194 _animatedSprite.FlipH = nextFlip;
195
196 // Sincroniza el arma con la dirección del sprite principal.
197 if (animatedWeapon != null)
198 {
199 animatedWeapon.FlipH = nextFlip;
200 var xOffset = nextFlip ? WeaponOffsetLeft : WeaponOffsetRight;
201 animatedWeapon.Position = new Vector2((xOffset - 6), animatedWeapon.Position.Y);
202 area2D.Position = new Vector2((nextFlip ? -14 : 14), area2D.Position.Y);
203 }
204 }

241 public void attack()
242 {
243 // Salida temprana para minimizar trabajo si no se puede atacar.
244 if (equip_weapon == null || !equip_weapon.attack() || CargarBullet == null || _animatedSprite == null)
245 return;
246
247 animatedWeapon.Play(equip_weapon.Name);
248 manaActual -= equip_weapon.Cost;
249
250 // Aplica el retroceso visual del arma (estilo Nuclear Throne).
251 if (animatedWeapon != null)
252 equip_weapon.ApplyRecoil(animatedWeapon, _animatedSprite.FlipH);
253
254 // Si el arma es cuerpo a cuerpo, no se instancia una bala.
255 if (equip_weapon.IsMeelee)
256 return;

259 bulletInstance.GlobalPosition = _animatedSprite.GlobalPosition;
260 bulletInstance.Configure(equip_weapon.direction, equip_weapon.Bullet, equip_weapon.Damage);
261 GetTree().CurrentScene?.AddChild(bulletInstance);
