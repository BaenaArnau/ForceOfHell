134 private static string[] SplitCsv(string line)
135 {
136 var values = new List<string>();
137 var current = new StringBuilder();
138 var insideQuotes = false;
139
140 foreach (var c in line)
141 {
142 switch (c)
143 {
144 case '"':
145 insideQuotes = !insideQuotes;
146 break;
147 case ',' when !insideQuotes:
148 values.Add(current.ToString().Trim());
149 current.Clear();
150 break;
151 default:
152 current.Append(c);
153 break;
154 }
155 }
156
157 values.Add(current.ToString().Trim());
158 return values.ToArray();
159 }
